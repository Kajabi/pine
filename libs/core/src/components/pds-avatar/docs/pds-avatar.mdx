import {DocArgsTable, DocCanvas} from '@pine-ds/doc-components';

# Avatar

Avatar is used for showing a thumbnail representation of an admin, customer, or business.

## Properties

<DocArgsTable componentName="pds-avatar" />

## Default

<DocCanvas client:only
  mdxSource={{
    react: '<PdsAvatar></PdsAvatar>',
    webComponent: '<pds-avatar></pds-avatar>'
}}>
  <pds-avatar></pds-avatar>
</DocCanvas>

### Admin

An admin variant of the avatar with less-rounded borders.

<DocCanvas client:only
  mdxSource={{
    react: '<PdsAvatar variant="admin"></PdsAvatar>',
    webComponent: '<pds-avatar variant="admin"></pds-avatar>'
}}>
  <pds-avatar variant="admin"></pds-avatar>
</DocCanvas>

### Badge

An avatar with a badge.

<DocCanvas client:only
  mdxSource={{
    react: '<PdsAvatar badge={true}></PdsAvatar>',
    webComponent: '<pds-avatar badge="true"></pds-avatar>'
}}>
  <pds-avatar badge="true"></pds-avatar>
</DocCanvas>


### Custom Sizes

An avatar with a custom size set on the prop.

<DocCanvas client:only
  mdxSource={{
    react: '<PdsAvatar size="128px"></PdsAvatar>',
    webComponent: '<pds-avatar size="128px"></pds-avatar>'
}}>
  <pds-avatar size="128px"></pds-avatar>
</DocCanvas>

### Dropdown

Allows the avatar to be used as a dropdown trigger.

<DocCanvas client:only
  mdxSource={{
    react: '<PdsAvatar dropdown={true}></PdsAvatar>',
    webComponent: '<pds-avatar dropdown="true"></pds-avatar>'
}}>
  <pds-avatar dropdown="true"></pds-avatar>
</DocCanvas>

### Image

An avatar using a user profile image. When not set, will default to the user icon.

<DocCanvas client:only
  mdxSource={{
    react: '<PdsAvatar alt="Customer Profile" image="../assets/demi_wilkinson.jpg"></PdsAvatar>',
    webComponent: '<pds-avatar alt="Customer Profile" image="../assets/demi_wilkinson.jpg"></pds-avatar>'
}}>
  <pds-avatar alt="Customer Profile" image="../assets/demi_wilkinson.jpg"></pds-avatar>
</DocCanvas>
