import { ArgsTable, Canvas, IconGallery, IconItem, Meta, Story } from '@storybook/addon-docs';
import {html, render} from 'lit-html';

import { extractArgTypes } from '@pxtrn/storybook-addon-docs-stencil';
import sageIconsJson from '../../../../icons/dist/sage-icons.json';

export const argTypes = () => {
  const extractedArgs = extractArgTypes('sage-icon');
  extractedArgs.name.control.type = 'select';
  extractedArgs.name.options = Object.values(sageIconsJson["icons"]).map((icon) => (icon.name));
  return extractedArgs;
}

<Meta title="components/Icon" component="sage-icon" argTypes={argTypes()}/>

export const Template = (args) => html`<sage-icon name=${args.name} size=${args.size}></sage-icon>`;

# Sage icon
Displays Icons used with the Sage Design System

## Guidelines

Icons are an effective way to enforce the purpose of an experience. Icons are broken out into two main
categories: Actions and Features. In order to ensure that the icons are adding to the clarity of an experience
and not creating confusion, we adhere to two principles.

## Properties

<ArgsTable of="sage-icon" />

### name
The `name` property is the name of the svg filename.

### size
The `size` property includes several pre-defined values for ease of use:

```
<sage-icon name="upload" size="small"></sage-icon>
<sage-icon name="upload"></sage-icon>
<sage-icon name="upload" size="medium"></sage-icon>
<sage-icon name="upload" size="large"></sage-icon>
```

 will render the following output:
 <Canvas withSource="none">
  <sage-icon name="upload" size="small"></sage-icon>
  <sage-icon name="upload" size="normal"></sage-icon>
  <sage-icon name="upload" size="medium"></sage-icon>
  <sage-icon name="upload" size="large"></sage-icon>
</Canvas>

It also accepts any custom size attribute. For example:
```
<sage-icon name="upload" size="60px"></sage-icon>
<sage-icon name="upload" size="2rem"></sage-icon>
```

will render the following output:
<Canvas withSource="none">
  <sage-icon name="upload" size="60px"></sage-icon>
  <sage-icon name="upload" size="2rem"></sage-icon>
</Canvas>
